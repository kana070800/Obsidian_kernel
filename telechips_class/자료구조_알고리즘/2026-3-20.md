BFS 조건 
--
1. 시작점 
	1. 외곽지점 >  0, N+1 
2. 인접 
	1. 상하좌우 
3. 종료점 
	1. 더이상 큐가 없을때
		1. 최솟값을 더이상 발견하지 못할때 
		2. 정상지점 도착하면 
4. 큐 
	1. 좌표 정보(x, y) 
5. 방문 표시 
	1. Cost MAP cost 인큐 조건으로 대체 

실제 코드조건
---
1. 필요한 변수
	1. Cost MAP 
		1. type: int 
	2. 입력 변수 
		1. 도착 좌표 저장 변수 
			1. int 2개 R, C
		2. N: 맵 크기 
			1. type: int 
		3. 높이 저장 MAP 
			1. type: int 
		4. 노드 
			1. 방문할 좌표 저장 r,c 
		5. 큐 
			1. 요소: NODE 
			2. 크기: `MAX_N^2*10 
		6. 상대 방향 저장용 변수(상하좌우) 
			1. nR, nC 
2. 초기화 방법 
	1. Cost MAP 초기화 > 0x7FFFFFFF
	2. 나머진 전역 0 초기화 
3. BFS 동작 
	1. 시작점(테두리)들에 대해 모두 인큐 
	2. 탐색 동작 실행 
		1. 디큐 
		2. 디큐 대상이 산정상이라면 continue 
		3. 산 정상이 아니라면 상하좌우 탐색 
			1. 경계 검사 (1~N) 
			2. 각 조건별로 COST 비용을 검사해서 이전보다 작으면 
				1. COST MAP 갱신 
				2. 해당 좌표 인큐 
	3. 더이상 큐가 없으면 산 정상 좌표 COST 값 리턴

#if 01

#define MAX_N (100+2)
int map[MAX_N][MAX_N];
int visited[MAX_N][MAX_N];
int N;
int R, C;

typedef struct _node {
	int r, c;
}node;
int front, rear;
node Queue[MAX_N * MAX_N*10];
void Enqueue(node data) {
	Queue[rear++] = data;
}

node Dequeue(void) {
	return Queue[front++];
}

void inputData(void) {
	front = rear = 0;

	scanf("%d %d %d", &N, &R, &C);

	for (int i = 0; i < MAX_N; i++)
		for (int j = 0; j < MAX_N; j++)
			visited[i][j] = 0x7fffffff;

	for (int i = 1; i <= N; ++i) {
		for (int j = 1; j <= N; ++j) {
			scanf("%d", &(map[i][j]));
			if ((i == 1) || (i == N) || (j == 1) || (j == N)) {
				Enqueue((node) { i, j});

				visited[i][j] = map[i][j] * map[i][j];
			}
		}
	}
}
int dR[4] = { -1, 1, 0, 0 };
int dC[4] = { 0, 0, -1, 1 };



int BFS(void) {
	int nR, nC;
	node curr;

	while (front != rear) {				//q에 존재시 반복
		curr = Dequeue();

		if ((curr.r == R) && (curr.c == C))
			continue;
		for (int i = 0; i < 4; i++)
		{     //인접점 탐색 후 존재시 enqueue 및 방문표시
			nR = curr.r + dR[i];
			nC = curr.c + dC[i];

			if ((nR > 0) && (nC > 0) && (nR <= N) && (nC <= N))
			{ //경계 만족
				int next = map[nR][nC];
				int current = map[curr.r][curr.c];
				int dcost;


				if (current == next)
				{ // 같다면 
					if (visited[nR][nC] > visited[curr.r][curr.c])
					{
						visited[nR][nC] = visited[curr.r][curr.c];
						Enqueue((node) { nR, nC });
					}
				}
				if (current < next)
				{  //크다면 바로 등산(enq)
					dcost = (next - current) * (next - current);
					if (visited[nR][nC] > visited[curr.r][curr.c] + dcost)
					{
						visited[nR][nC] = visited[curr.r][curr.c] + dcost;
						Enqueue((node) { nR, nC });
					}
				}
				if (current > next)
				{  //작다면 바로 등산(enq)
					dcost = (current - next);
					if (visited[nR][nC] > visited[curr.r][curr.c] + dcost)
					{
						visited[nR][nC] = visited[curr.r][curr.c] + dcost;
						Enqueue((node) { nR, nC });
					}
				}
			}
		}
	}
	return visited[R][C];
}

}
