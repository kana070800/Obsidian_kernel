	탐색을 위해 linked-list 데이터를 배열에 저장하여 참조하는 식으로 활용가능

typedef int ElementType;
typedef struct _node {
	ElementType data;
	struct _node* next;

}SLL_node;

void print_SLLnode(SLL_node * Head) {
	SLL_node* curr;
	for (curr = Head; curr != NULL; curr = curr->next) {
		printf("%d ", curr->data);
	}

} //노드 data 프린트

SLL_node* SLL_CreateNode(ElementType *in_data) {
	SLL_node* new_node;
	new_node = (SLL_node*)calloc(1, sizeof(SLL_node));

	if (new_node == NULL)
		return NULL;

	new_node->data = *in_data;
	new_node->next = NULL;

	return new_node;
} // 하나의 노드를 생성, 주소 반환

void SLL_AppendNode(SLL_node** Head, SLL_node* new_node) {
	if (Head == NULL) return;

	SLL_node* curr = *Head;
	
	if (curr == NULL) {
		*Head = new_node;
	}
	else {
		while (curr->next) {
			curr = curr->next;
		}
		curr->next = new_node;

	}
	return;
}  // 노드 주소를 받아 append, tail이 없을 때

void SLL_AppendNode1(SLL_node** Head, SLL_node** Tail, SLL_node* new_node) {
	if (Head == NULL) return;


	if (*Head == NULL) {
		*Head = new_node;
		*Tail = new_node;
	}
	else {
		(*Tail)->next = new_node;
		*Tail = new_node;
	}
	return;
} // 노드 주소를 받아 append, tail 존재
