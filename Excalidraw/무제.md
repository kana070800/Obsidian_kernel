big endian
	하위 bit를 상위 주소에 저장
	사람 친화적
		낮은 주소부터 읽어서 msb로 배치하는 작업 필요
little endian
	상위 bit를 상위 주소에 저장
	프로그램 친화적
		시작주소에서 그대로 읽어오면 된다

	컴파일 시 option으로 선택가능

scatter loading 예시
```
/* 1. 메모리 영역 정의 */
MEMORY
{
    FLASH (rx) : ORIGIN = 0x08000000, LENGTH = 128K
    RAM   (rwx): ORIGIN = 0x20000000, LENGTH = 20K
}

/* 2. 섹션 배치 정의 */
SECTIONS
{
    /* 코드 섹션: FLASH에 배치 */
    .text :
    {
        . = ALIGN(4);      /* 4바이트 정렬 */
        *(.vectors)        /* 인터럽트 벡터 테이블 우선 배치 */
        *(.text)           /* 모든 .text 섹션 */
        *(.rodata)         /* 읽기 전용 데이터 */
        . = ALIGN(4);
    } > FLASH

    /* 초기화된 데이터 섹션: 
       LMA(로드 주소)는 FLASH, VMA(가상 주소)는 RAM */
    .data : 
    {
        . = ALIGN(4);
        _sdata = .;        /* 데이터 시작 심볼 */
        *(.data)
        . = ALIGN(4);
        _edata = .;        /* 데이터 끝 심볼 */
    } > RAM AT > FLASH      /* 핵심: RAM에 배치되지만 FLASH에 저장됨 */

    /* 초기화되지 않은 데이터 섹션 */
    .bss :
    {
        _sbss = .;
        *(.bss)
        *(COMMON)
        . = ALIGN(4);
        _ebss = .;
    } > RAM
}
```
